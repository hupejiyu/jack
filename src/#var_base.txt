    _max_iter = args[1]

    $_tmp = ''

    _counter = 0
    :_vb_full_cycle_loop
        !dynamic "<<$_arr_name>>[<<_counter>>] = dyneval($vb_cycle, <<_counter>>, <<$_caption>>)"

    killvar '$_arr_name'
    killvar '_max_iter'
    killvar '$_caption'
    killvar '_counter'
    killvar '$_tmp'
